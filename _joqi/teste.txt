select	pai.nmPessoa,
		filho1.nmPessoa,
		filho2.nmPessoa
from 	pessoas pai,
		pessoas filho1,
		pessoas filho2
where   filho1.cdPessoa = 2 and		
		filho1.vlIdade between 20 and 30 and
		filho1.cdPessoa <> filho2.cdPessoa and
		filho1.nmPessoa like '%douglas%' and
		filho1.cdPessoa > filho2.cdPessoa or		 
		not filho1.apareceNaConsulta is false
