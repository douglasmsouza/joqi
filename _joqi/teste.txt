select	avo.nmPessoa avo,
		pai.nmPessoa pai,			
		mae.nmPessoa mae,
		filho.nmPessoa filho		
from 	pessoas avo,
		pessoas pai,
		pessoas mae,
		pessoas filho
where	filho.cdPai = pai.cdPessoa and
		pai.cdPai = avo.cdPessoa and
		mae.cdPessoa = filho.cdMae		
order by avo.nmPessoa, 
		 pai.nmPessoa, 
		 mae.nmPessoa, 
		 filho.nmPessoa
